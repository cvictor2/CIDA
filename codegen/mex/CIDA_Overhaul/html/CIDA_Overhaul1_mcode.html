1%function Chafee_Infante(N1,T,nui,alphai,graph)
  2%Chafee-Infante eq. u_t - nu*u_xx = u - alpha*u^3
  3%basic constants defined by problem
  4
  5%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  6%Solves Chafee-Infante equation using data assimilation on a uniform grid
  7%
  8%Meant to use loaded data from Data directory
  9%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 10% close all; clear all;
 11close all;
 12clear all;
 13% param = struct; %All of the parameters for this problem: alpha, nu, min_nodes, L, N, mu
 14
 15%% Load Variables
 16% load('Data\2018.01.23.01.24.03.575.mat')
 17% load('Data/Overhauled/2018.04.10.23.18.30_Initial_Ramp_Only_nu=1.0e-04_alpha=1_mu=100.mat')
 18
 19graph = true;           %Display Graphs
 20save_check = false;     %Save data after ramp up and at end
 21reset_vars = true;      %Override loaded variable configurations with new definitions
 22if( exist('param','var') == 0)   
 23    alpha = 1;
 24    nu = 7.5e-6;
 25    %     nu is in [7.5e-6,0.01]
 26    
 27    L = 1;
 28    N = 2^12;
 29    mu = 200;
 30    min_nodes = max(ceil(.25*nu^-.5),5);
 31    
 32    seed = randi(10000)
 33%          seed = 7263
 34         T_ramp = 10;
 35    T = 15;
 36    %     param = struct('alpha',alpha,'nu',nu,'L',L,'mu',mu,'min_nodes',min_nodes,'seed',seed,'T',T);
 37    
 38    
 39    %% Initialize other constants
 40    
 41    dx = L/N;
 42    %x = linspace(0,L,N);
 43    x = 0 + dx*(0:(N-1));
 44    t_ramp = 0;
 45    
 46    dt = .001; %0.001;
 47    timesteps = ceil(T/dt);
 48    
 49    
 50    %% Initial Conditions
 51    % seed = randi(10000);
 52    % fprintf('seed: %d\n', seed);
 53    rng(seed);
 54    a = (10^(0))*randn(1,floor(N/5));
 55    k = 1:1:floor(N/5);
 56    u_0 = zeros(1,N);
 57    for i = 1:floor(N/5)
 58        u_0 = u_0 + (a(i)*sin(2*pi*k(i)*x/L));
 59    end
 60    u_0 = u_0/(norm(u_0)*sqrt(dx))*10^-3;
 61    
 62    % dt = dx/(1/sqrt(alpha)); % CFL, Courant Fredrich Lewy number for advection
 63
 64    
 65    
 66    u = u_0';
 67    ui = u(2:N);
 68    t=0;
 69    
 70    %% Initialize Param
 71    param = struct('alpha',alpha,'nu',nu,'L',L, 'N',N,'mu',mu,'min_nodes',min_nodes,'seed',seed,'T',T, 't', t, 'dt', dt, 'x', x, 'dx', dx, 'u_0', u_0, 't_ramp',t_ramp,'t_adj',t-t_ramp,'u_ramp',u_0, 'timesteps',timesteps);
 72    
 73    %% Ramp up
 74    offset = 0;
 75    u_solo = struct('u',u,'type',0);
 76
 77    % Convex Splitting Method
 78    A = (nu/(dx^2))*gallery('tridiag',N-1,1,-2,1) + (1+2*alpha)*speye(N-1);
 79    B = speye(N-1) - dt*A;
 80    
 81
 82    while(max(abs(u))<.8/sqrt(alpha)&&t<T_ramp-dt)
 83        
 84        if (graph && mod(offset,200) == 0)
 85            if(~Graphing_Overhaul(param,u_solo,[]))
 86                return
 87            end
 88            %     pause;
 89        end
 90        
 91        %solves pde
 92        offset = offset +1;
 93        t = t+dt;
 94        param.t = t;
 95%         param.t_adj = t - t_ramp;
 96
 97        %     umv = u - v;
 98        %error_DA(k) = sqrt(dx)*norm(umv);
 99        ui = B\(ui.*(ones(N-1,1)+dt.*(alpha.*(-2-(ui.^2))))); %Convex Splitting Method
100        u(2:N) = ui;
101        u_solo.u = u;
102%         u_f = abs(fft(u)/N);
103        %     umv_error(offset) = sqrt(dx)*norm(u-v);
104    end
105    t_ramp = t;
106    param.t_ramp = t_ramp;
107    param.t_adj = t-t_ramp;
108    u_ramp = u;
109    param.u_ramp = u_ramp;
110    
111    if(save_check)
112        date_string = datestr(now,'yyyy.mm.dd.HH.MM.SS');
113        save(sprintf('Data/Overhauled/%s_Initial_Ramp_Only_nu=%.1d_alpha=%.0d_mu=%.0d.mat',date_string,nu,alpha,mu),'param');
114    end
115    vars = [];
116else
117    %% Restore from loaded variables
118    alpha = param.alpha;
119    nu = param.nu;
120    L = param.L;
121    N = param.N;
122    mu = param.mu;
123    min_nodes = param.min_nodes;
124    
125    seed = param.seed;
126    T = param.T;
127    t = param.t;
128    
129    x = param.x;
130    dx = param.dx
131    t_ramp = param.t_ramp;
132    
133    rng(seed);
134    
135    u_0 = param.u_0;
136    u_ramp = param.u_ramp;
137    u = u_ramp;
138    u_solo = struct('u',u_ramp,'type',0);
139    
140    dt = param.dt;
141    timesteps = param.timesteps;
142    
143    offset = t_ramp/dt;
144
145    if(graph)
146        if(~Graphing_Overhaul(param,u_solo,[]))
147            return
148        end
149    end
150    
151    % Convex Splitting Method
152    A = (nu/(dx^2))*gallery('tridiag',N-1,1,-2,1) + (1+2*alpha)*speye(N-1);
153    B = speye(N-1) - dt*A;
154    
155end
156
157%% Set up variables to track
158offset = 0; %comment this out is you only want to run to time T, not time t_ramp + T
159if(( exist('vars','var') == 0)||reset_vars)
160    %Type Descriptions:
161    %   0 - Normal solution, no data assimilation
162    %   1 - Uniform Grid Data Assimilation
163    %   2 - Standard Car Data Assimilation
164    %  -1 - Retro Car, old car method
165    %   3 - Car with uniform grid along length (Hybrid Method)
166    vars = [];
167    vars = repelem(struct,1,1);
168    i = 0;
169        % Standard uniform grid setup
170    i = i+1;
171    vars(i).type = 1;
172    vars(i).v = zeros(size(u));
173    vars(i).int_nodes = min_nodes;
174    vars(i).i_nodes = floor(linspace(1,N-1,vars(i).int_nodes));
175    vars(i).x_nodes = x(vars(i).i_nodes);
176    vars(i).error = zeros(1,timesteps-offset+1);
177
178    % Standard car setup
179    i = i+1;
180    vars(i).type = 2;
181    vars(i).v = zeros(size(u));
182    vars(i).int_nodes = 30;
183    vars(i).i_nodes = 1:vars(i).int_nodes;
184    vars(i).x_nodes = x(vars(i).i_nodes);
185    vars(i).velocity = 10;
186    vars(i).direction = 1;
187    vars(i).error = zeros(1,timesteps-offset+1);
188
189    
190    
191    %Templates (Don't adjust these, just make copies if you want to use an
192    %           instance)
193%     % Standard uniform grid setup
194%     i = i+1;
195%     vars(i).type = 1;
196%     vars(i).v = zeros(size(u));
197%     vars(i).int_nodes = min_nodes;
198%     vars(i).i_nodes = floor(linspace(1,N-1,vars(i).int_nodes));
199%     vars(i).x_nodes = x(vars(i).i_nodes);
200%     vars(i).error = zeros(1,timesteps-offset+1);
201%     
202%     % Standard car setup
203%     i = i+1;
204%     vars(i).type = 2;
205%     vars(i).v = zeros(size(u));
206%     vars(i).int_nodes = 30;
207%     vars(i).i_nodes = 1:vars(i).int_nodes;
208%     vars(i).x_nodes = x(vars(i).i_nodes);
209%     vars(i).velocity = 10;
210%     vars(i).direction = 1;
211%     vars(i).error = zeros(1,timesteps-offset+1);
212% 
213%     %Retro Car Configuration
214%     i = i+1;
215%     vars(i).type = -1;
216%     vars(i).v = zeros(size(u));
217%     vars(i).length_car = 10;
218%     vars(i).int_nodes = vars(i).length_car;
219%     vars(i).i_nodes = 1:vars(i).int_nodes;
220%     vars(i).x_nodes = x(vars(i).i_nodes);
221%     vars(i).velocity = vars(i).length_car;
222%     vars(i).direction = 1;
223%     vars(i).error = zeros(1,timesteps-offset+1);
224    
225%     %Uniform Grid on Car Configuration
226%     i = i+1;
227%     vars(i).type = 3;
228%     vars(i).v = zeros(size(u));
229%     vars(i).length_car = 10; %Length of car (in gridpoints)
230%                              %Set to N to cover entire grid
231%     vars(i).int_nodes = 2;
232%     vars(i).i_nodes = floor(linspace(1,vars(i).length_car, vars(i).int_nodes));
233%     vars(i).x_nodes = x(vars(i).i_nodes);
234%     vars(i).velocity = 10;
235%     vars(i).direction = 1;
236%     vars(i).error = zeros(1,timesteps-offset+1);
237    clear i;
238end
239
240% P = parallel.pool.Constant(param);
241% P.Value.
242%% Solve System
243for j = 1:timesteps-offset
244    t = t+dt;
245    param.t = t;
246    param.t_adj = param.t_adj + dt;
247%     U = parallel.pool.Constant(u);
248%     U.Value;
249    %update the v's
250    for (k = 1:length(vars))
251        vars(k).error(j) = sqrt(dx)*norm(u-vars(k).v);
252        switch vars(k).type
253            case {2,-1, 3}
254                %update car nodes
255                vars(k) = Update_Overhaul(param, vars(k));
256        end
257        umv = u - vars(k).v;
258        Ih_umv = LinTerp(param, vars(k), umv);
259        vars(k).v(2:N) = B\(vars(k).v(2:N)+dt*(alpha*(-2*vars(k).v(2:N)-(vars(k).v(2:N).^3))+ (mu)*Ih_umv)); %Convex Splitting Method
260    end
261    %update u
262    u(2:N) = B\(u(2:N).*(ones(N-1,1)+dt.*(alpha.*(-2-(u(2:N).^2))))); %Convex Splitting Method
263    u_solo.u = u;
264    
265    show = 100;
266% show = 1;
267    if(j<=50)
268        show = 1;
269    end
270    %     if(max(u)> 0.9/sqrt(alpha)&&graph && mod(k,5) == 0
271    if(graph && mod(j,show) == 0)
272        if(~Graphing_Overhaul(param,u_solo,vars))
273            return
274        end
275    end
276end
277%Get last error measurement
278 for k = 1:length(vars)
279        vars(k).error(end) = sqrt(dx)*norm(u-vars(k).v);
280 end
281 
282 %% Plot Error
283 figure
284 legendInfo = cell(1,length(vars));
285 for k = 1:length(vars)
286     if(k==2)
287         semilogy(0:dt:T,vars(k).error,'--','LineWidth',1);
288     else
289         semilogy(0:dt:T,vars(k).error,'LineWidth',1);
290     end
291     hold on;
292     switch vars(k).type
293         case 0
294             text = 'u';
295         case 1
296             text = 'Uniform Grid';
297         case 2 
298             text = 'Sweeping Probe';
299         otherwise
300             text = 'Misc.';
301     end
302     legendInfo{k} = [text];
303 end
304legend(legendInfo);
305xlabel('Time');
306ylabel('L^2 Norm of u-v');
307title('Error Plot for Different Methods');
308% date = datestr(now,'yyyy.mm.dd.HH.MM.SS');
309if(save_check)
310    if(~exists(date_string))
311        date_string = datestr(now,'yyyy.mm.dd.HH.MM.SS');
312    end
313    save(sprintf('Data/Overhauled/%s_Initial_Ramp_Var_Config_nu=%.1d_alpha=%.0d_mu=%.0d.mat',date_string,nu,alpha,mu),'param','vars');
314end

317function [lin] = LinTerp(p,v,umv)
318switch v.type
319    case 1
320        lin = interp1(v.x_nodes,umv(v.i_nodes),p.x)';
321        lin(end) = [];
322    case {2,-1, 3}
323        [maximum,index] = max(v.x_nodes(2:end)-v.x_nodes(1:end-1));
324        if(maximum*p.dx) > ceil((p.L/v.int_nodes)*p.dx)
325%             maximum
326            group1 = struct('i_nodes',v.i_nodes(1:index),'x_nodes',v.x_nodes(1:index));
327            group2 = struct('i_nodes',v.i_nodes(index+1:end),'x_nodes',v.x_nodes(index+1:end));
328            if(index ==1)
329                lin = interp1(group2.x_nodes,umv(group2.i_nodes),p.x,'linear',0)';
330                lin(1) = umv(1);
331            elseif(length(group2.i_nodes)==1)
332                lin = interp1(group1.x_nodes,umv(group1.i_nodes),p.x,'linear',0)';
333                lin(p.N-1) = umv(p.N-1);
334            elseif(v.i_nodes(1) ==0) %0 is included
335                lin1 = interp1(group1.x_nodes(2:end),umv(group1.i_nodes(2:end)),p.x,'linear',0)';
336                lin2 = interp1(group2.x_nodes,umv(group2.i_nodes),p.x,'linear',0)';
337                lin = lin1 + lin2;
338
339
340            else %0 is not included
341                lin1 = interp1(group1.x_nodes,umv(group1.i_nodes),p.x,'linear',0)';
342                lin2 = interp1(group2.x_nodes,umv(group2.i_nodes),p.x,'linear',0)';
343                lin = lin1 + lin2;
344            end
345            lin(end) = [];
346
347        else
348            %No gap in the car
349            if(v.i_nodes(1) ==0)
350                lin = interp1(v.x_nodes(2:end),umv(v.i_nodes(2:end)),p.x,'linear',0)';
351            else
352                lin = interp1(v.x_nodes,umv(v.i_nodes),p.x,'linear',0)';
353
354
355            end
356            lin(end) = [];
357        end
358    otherwise
359        lin = zeros(p.N-1,1);
360        
361end
362% lin = 1;
363end

365function [var] = Update_Overhaul(p, v)
366% var = v;
367% v.x_nodes = unique(floor(mod(v.x_nodes + v.velocity*p.t_adj,1)./p.dx).*p.dx);
368switch v.type
369    case -1
370        v.i_nodes = unique(mod(v.i_nodes+v.direction*v.velocity,p.N));
371        if(v.i_nodes(1)==0)
372            v.i_nodes(1) = [];
373            v.i_nodes = [v.i_nodes, p.N];
374        end
375        v.x_nodes = p.x(v.i_nodes);
376    case 2
377        v.x_nodes = unique(floor(mod(v.x_nodes + v.direction*v.velocity*p.dt,1)./p.dx).*p.dx);
378        [maximum,index] = max(v.x_nodes(2:end)-v.x_nodes(1:end-1));
379        if(maximum > p.dx*v.int_nodes)
380            v.i_nodes(1:index) = v.i_nodes(1:index);
381        elseif(maximum == p.dx*2) %Gap caused when crossing over boundary
382            v.x_nodes(1:index) = v.x_nodes(1:index) - p.dx;
383            v.i_nodes = v.x_nodes./p.dx;
384            
385        else
386            v.i_nodes = v.x_nodes./p.dx;
387        end    
388    case 3
389        v.x_nodes = unique(floor(mod(v.x_nodes + v.direction*v.velocity*p.dt,1)./p.dx).*p.dx);
390         v.i_nodes = v.x_nodes./p.dx;
391         if(v.i_nodes(1)==0)
392             v.i_nodes(1) = [];
393             v.i_nodes = [v.i_nodes, p.N];
394             v.x_nodes = p.x(v.i_nodes);
395         end
396    otherwise
397        %         var = v;
398end
399var = v;
400end
